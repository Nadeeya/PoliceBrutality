from datetime import datetime
def convert(date_str):
    date_obj = datetime.strptime(date_str , '%Y-%m-%d')
    return date_obj
date = list(df.date)
for d in date:
    d_obj = convert(d)
    df.date.replace(to_replace=d , value=d_obj.year , inplace=True)
